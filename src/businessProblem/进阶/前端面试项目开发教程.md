项目概述
这是一个基于 Gulp 构建的前端面试学习项目，包含多个面试知识点的实践模块。项目采用模块化开发，每个模块都是一个独立的学习主题，涵盖 HTML、CSS、JavaScript、HTTP、安全等前端核心知识点。
技术栈
● 构建工具: Gulp 3.9.1
● 前端技术: HTML5 + CSS3(Less) + ES6
● 后端服务: Express.js + Node.js
● 开发工具: Browser-sync (热重载)
● 代码转换: Babel (ES6转ES5)
● 样式处理: Less + Autoprefixer + CSSnano
● 图片优化: Imagemin + Pngquant

gulp.task('server', function() {
  var server_opts = null;
  
  switch (env) {
    case "web_proxy":
      // 前后端开发模式
      server_opts = {
        proxy: 'http://localhost:' + conf.api_port,
        browser: 'chrome',
        notify: false,
        port: conf.dev_port
      };
      break;
    case "web_only":
      // 前端开发模式
      server_opts = {
        notify: false,
        server: path.resolve(__dirname, opts.dist),
        index: './index.html',
        port: conf.dev_port,
        logConnections: true
      };
      break;
  }
  
  browserSync.init(server_opts);
  
  // 文件监听
  gulp.watch(path.join(opts.one, '*.html'), ['html']);
  gulp.watch(path.join(opts.one, '*.less'), ['less']);
  gulp.watch(path.join(opts.one, '*/*.{png,jpg,gif,svg}'), ['images']);
  gulp.watch(path.join(opts.one, '*.js'), ['js']);
});

// HTML处理任务
gulp.task('html', function() {
  var options = {
    removeComments: true,
    collapseWhitespace: true,
    collapseBooleanAttributes: false,
    removeAttributeQuotes: false,
    removeEmptyAttributes: false,
    removeScriptTypeAttributes: true,
    removeStyleLinkTypeAttributes: true,
    minifyJS: true,
    minifyCSS: true
  };
  
  return gulp.src(path.join(opts.one, '*.html'))
    .pipe(plumber())
    .pipe(htmlmin(options))
    .pipe(gulp.dest(opts.one_dist))
    .pipe(reload({ stream: true }));
});

// Less编译任务
gulp.task('less', function() {
  return gulp.src(path.join(opts.one, '*.less'))
    .pipe(plumber())
    .pipe(less())
    .pipe(cssnano())
    .pipe(autoprefixer({
      browsers: ['last 2 versions', '>5%', 'Firefox >= 20', 'Chrome >=40'],
      cascade: false
    }))
    .pipe(gulp.dest(opts.one_dist))
    .pipe(reload({ stream: true }));
});

// 图片压缩任务
gulp.task('images', function() {
  return gulp.src(path.join(opts.one, '*/*.{png,jpg,gif,svg}'))
    .pipe(plumber())
    .pipe(changed(opts.one_dist))
    .pipe(imagemin({
      progressive: true,
      svgoPlugins: [{ removeViewBox: false }],
      use: [pngquant()]
    }))
    .pipe(gulp.dest(opts.one_dist))
    .pipe(reload({ stream: true }));
});

// JavaScript处理任务
gulp.task('js', function() {
  return gulp.src(path.join(opts.one, '*.js'))
    .pipe(plumber())
    .pipe(uglify())
    .pipe(gulp.dest(opts.one_dist))
    .pipe(reload({ stream: true }));
});

// 默认任务
gulp.task('default', ['server'], function() {
  gulp.start(['html', 'less', 'js', 'images']);
});

var FN = {
  // Cookie操作
  getCookie: function(c_name) {
    if (document.cookie.length > 0) {
      c_start = document.cookie.indexOf(c_name + "=");
      if (c_start != -1) {
        c_start = c_start + c_name.length + 1;
        c_end = document.cookie.indexOf(";", c_start);
        if (c_end == -1) c_end = document.cookie.length;
        return unescape(document.cookie.substring(c_start, c_end))
      }
    }
    return ""
  },
  
  setCookie: function(c_name, value, expiredays, path) {
    var exdate = new Date();
    exdate.setDate(exdate.getDate() + expiredays);
    document.cookie = c_name + "=" + escape(value) + 
      ((expiredays == null) ? "" : ";expires=" + exdate.toGMTString()) + 
      ((path == null) ? "" : ";path=" + path);
  },
  
  // URL参数获取
  getParam: function(name) {
    var reg = new RegExp("(^|&)" + name + "=([^&]*)(&|$)", "i");
    var r = window.location.search.substr(1).match(reg);
    if (r != null) return unescape(r[2]);
    return null;
  },
  
  // 时间格式化
  formatterDateDay: function(date, flag) {
    if (!date) return false;
    var dt = new Date(date);
    if (flag && flag == true) {
      return (dt.getFullYear() + "-" + this.reset_Num(dt.getMonth() + 1) + "-" + this.reset_Num(dt.getDate()));
    } else {
      return (dt.getFullYear() + "-" + this.reset_Num(dt.getMonth() + 1) + "-" + this.reset_Num(dt.getDate()) + 
        " " + this.reset_Num(dt.getHours()) + ":" + this.reset_Num(dt.getMinutes()) + ':' + this.reset_Num(dt.getSeconds()));
    }
  },
  
  reset_Num: function(num) {
    if (num < 10) {
      return "0" + num;
    }
    return num;
  },
  
  // AJAX封装
  ajax: function(obj) {
    var opts = {
      url: conf.root_path + obj.url,
      dataType: "json",
      type: "POST",
    };
    
    if (obj.data != undefined) {
      opts.data = obj.data;
    }
    
    return $.ajax(opts);
  },
  
  // 表单验证
  checkPhone: function(phone, id) {
    var mobile_r = /^1[3|4|5|7|8]\d{9}$/;
    var phone_r = /^0\d{2,3}-?\d{7,8}$/;
    
    if (mobile_r.test(phone) || phone_r.test(phone)) {
      return true;
    } else {
      return false;
    }
  }
};

function App() {
  var me = this;
  me.padding = 5;
  me.box_num = 600;
  me.obj = {
    ac_dom: null,
    ac_index: -1,
    go_index: -1,
    ac_x_mid: 0,
    ac_y_mid: 0,
    ac_dom_x: 0,
    ac_dom_y: 0,
    x_0: 0,
    y_0: 0,
  };
  me.load_key = false;
  me.box_data = [];
  me.img_style_obj = {};
}

App.prototype = {
  init: function() {
    var me = this;
    me._bind();
    me._init();
    me._init_style();
    me._item_pos();
    me._item_ev();
    me._start();
  },
  
  _bind: function() {
    var me = this;
    // 绑定所有方法到实例
    var fns = {
      _init: function() {
        me.box_w = me._get_num();
        me.box_h = me._get_num();
        me.img_w = me.box_w - me.padding * 2 * 3;
        me.img_h = me.box_h - me.padding * 2 * 3;
      },
      
      _get_num: function() {
        var min = $("body").width() < $("body").height() ? $("body").width() : $("body").height();
        return min < me.box_num ? min : me.box_num;
      },
      
      _init_style: function() {
        // 初始化拼图样式和位置
        $('#box').css({
          width: `${me.box_w}px`,
          height: `${me.box_h}px`,
        });
        
        $('#box>.item').css('padding', `${me.padding}px`);
        $('#box>.item>.img').css('backgroundSize', `${me.img_w}px ${me.img_h}px`);
        
        var doms = $('#box>.item');
        me.box_data.length = 0;
        
        for (var index = 0; index < doms.length; index++) {
          $(doms[index]).css({
            top: `${Math.floor(index / 3)*me.box_h / 3}px`,
            left: `${index % 3*me.box_w / 3}px`,
          });
          
          // 收集盒子数据
          me.box_data.push({
            left_0: index % 3 * me.box_w / 3,
            left_1: index % 3 * me.box_w / 3 + me.box_w / 3,
            top_0: Math.floor(index / 3) * me.box_h / 3,
            top_1: Math.floor(index / 3) * me.box_h / 3 + me.box_h / 3,
          });
          
          // 设置背景图片位置
          $(doms[index]).children('.img').css({
            backgroundRepeat: 'no-repeat',
            backgroundPosition: `-${index % 3*me.img_w / 3}px -${Math.floor(index / 3)*me.img_h / 3}px`,
          });
          
          // 收集样式对象
          me.img_style_obj[index] = `
            background-size:${me.img_w}px ${me.img_h}px;
            background-repeat:no-repeat; 
            background-position:-${index % 3*me.img_w / 3}px -${Math.floor(index / 3)*me.img_h / 3}px;
          `;
        }
      },
      
      _item_pos: function() {
        // 设置拼图块位置和标识
        var doms = $('#box>.item');
        for (var index = 0; index < doms.length; index++) {
          $(doms[index])
            .css({
              top: `${Math.floor(index / 3)*me.box_h / 3}px`,
              left: `${index % 3*me.box_w / 3}px`,
            })
            .attr('_index', index)
            .attr('id', `item_${index}`);
            
          $(doms[index]).children('.img').attr('_index', index);
        }
      },
      
      _item_ev: function() {
        // 拖拽事件处理
        $('#box').on('mousedown', '.item', function(e) {
          me.ac_dom = e.currentTarget;
          $(me.ac_dom).addClass('ac');
          
          me.ac_dom_x = $(me.ac_dom).position().left;
          me.ac_dom_y = $(me.ac_dom).position().top;
          
          me.x_0 = e.clientX - $('#box').offset().left;
          me.y_0 = e.clientY - $('#box').offset().top;
        });
        
        $('#box').on('mousemove', function(e) {
          if (me.ac_dom == null) return;
          
          $(me.ac_dom).css({
            left: `${me.ac_dom_x+(e.clientX-$('#box').offset().left)-me.x_0}px`,
            top: `${me.ac_dom_y+(e.clientY - $('#box').offset().top)-me.y_0}px`,
          });
          
          me._item_matching();
        });
        
        $('#box').on('mouseup', '.item', function(e) {
          me._item_matched();
          me._item_over();
        });
      },
      
      _start: function() {
        // 打乱拼图
        var doms = $('#box>.item');
        var arr_index = [];
        for (var index = 0; index < doms.length; index++) {
          arr_index.push(index);
        }
        
        $('#start').on('click', function() {
          arr_index.sort(function(a, b) {
            return Math.random() > .5 ? -1 : 1;
          });
          
          for (var index = 0; index < doms.length; index++) {
            $(`#item_${index}`).children('.img').attr({
              '_index': arr_index[index],
              'style': me.img_style_obj[arr_index[index]]
            });
          }
        });
      },
      
      // 其他方法...
    };
    
    for (var fn in fns) {
      me[fn] = fns[fn];
    }
  },
};

# 页面布局

### 三栏布局实现方式

1. **浮动布局**
   - 优点：兼容性好，中间模块完全撑开
   - 缺点：脱离文档流，需要清除浮动

2. **绝对定位**
   - 优点：快捷
   - 缺点：可使用性差

3. **Flexbox布局**
   - 优点：解决上述问题，移动端友好
   - 缺点：IE兼容性

4. **表格布局**
   - 优点：兼容性非常好，单元格同时增高
   - 缺点：语义化不好

5. **网格布局**
   - 优点：代码简洁
   - 缺点：浏览器支持有限
